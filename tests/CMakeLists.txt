macro(alfc_test file)
  add_test(
    NAME ${file}
    COMMAND $<TARGET_FILE:alfc> ${CMAKE_CURRENT_LIST_DIR}/${file}
    WORKING_DIRECTORY ${CMAKE_SOURCE_DIR}
  )
  set_tests_properties(${file} PROPERTIES TIMEOUT 40)
endmacro()

foreach(file ${alfc_test_file_list})
  alfc_test(${file})
endforeach()
